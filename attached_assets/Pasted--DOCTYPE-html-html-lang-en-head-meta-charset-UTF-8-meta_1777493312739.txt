    <script>
        const imagePool = [
            "https://i.imgur.com/At5EvhS.png",
            "https://i.imgur.com/KuGij8t.png",
            "https://i.imgur.com/BLgerSI.png",
            "https://i.imgur.com/FhlxwGz.png",
            "https://i.imgur.com/Fz1ylKq.png"
        ];

        const btn = document.getElementById('spinBtn');
        const tickSound = document.getElementById('tickSound');
        const slotSound = document.getElementById('slotSound');
        const slotImages = [
            document.getElementById('slot1'),
            document.getElementById('slot2'),
            document.getElementById('slot3')
        ];
        const slotDivs = document.querySelectorAll('.slot');

        function easeOutCubic(t) {
            return 1 - Math.pow(1 - t, 3);
        }

        function playTick() {
            tickSound.pause();
            tickSound.currentTime = 0;
            tickSound.play().catch(() => {});
        }

        async function runSlot(index) {
            const imgElement = slotImages[index];
            const parentDiv = slotDivs[index];
            let lastRandomIndex = -1;
            
            parentDiv.classList.remove('locked');
            parentDiv.classList.add('active');

            const startTime = performance.now();
            const fastDuration = 2000;
            const slowDuration = 4000;
            const totalDuration = fastDuration + slowDuration;

            return new Promise(resolve => {
                function tick(currentTime) {
                    const elapsed = currentTime - startTime;

                    if (elapsed < totalDuration) {
                        let currentDelay;

                        if (elapsed < fastDuration) {
                            currentDelay = 60;
                        } else {
                            const slowProgress = (elapsed - fastDuration) / slowDuration;
                            currentDelay = 60 + (easeOutCubic(slowProgress) * 640);
                        }

                        let randomIndex;
                        do {
                            randomIndex = Math.floor(Math.random() * imagePool.length);
                        } while (randomIndex === lastRandomIndex);
                        
                        lastRandomIndex = randomIndex;
                        imgElement.src = imagePool[randomIndex];
                        
                        playTick();

                        setTimeout(() => {
                            requestAnimationFrame(tick);
                        }, currentDelay);
                    } else {
                        parentDiv.classList.remove('active');
                        parentDiv.classList.add('locked');
                        resolve();
                    }
                }
                requestAnimationFrame(tick);
            });
        }

        btn.addEventListener('click', async () => {
            // Force browser to unlock audio
            [tickSound, slotSound].forEach(s => {
                s.play().then(() => {
                    s.pause();
                    s.currentTime = 0;
                }).catch(() => {});
            });

            // Start the payout alarm
            slotSound.play().catch(() => {});

            btn.disabled = true;

            await runSlot(0);
            await runSlot(1);
            await runSlot(2);

            btn.disabled = false;
        });
    </script>
